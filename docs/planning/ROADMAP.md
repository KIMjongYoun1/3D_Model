# test_sam.py
from segment_anything import sam_model_registry, SamPredictor
import cv2

def test_sam():
    print("=== SAM í…ŒìŠ¤íŠ¸ ì‹œì‘ ===")
    
    # 1. ëª¨ë¸ ì²´í¬í¬ì¸íŠ¸ ë‹¤ìš´ë¡œë“œ (ìµœì´ˆ 1íšŒ)
    # https://github.com/facebookresearch/segment-anything#model-checkpoints
    # sam_vit_h_4b8939.pth ë‹¤ìš´ë¡œë“œ í•„ìš”
    
    checkpoint_path = "sam_vit_h_4b8939.pth"

        print("https://dl.fbaipublicfiles.com/segment_anything/sam_vit_h_4b8939.pth")

    # 2. ëª¨ë¸ ë¡œë“œ
    print("SAM ëª¨ë¸ ë¡œë”© ì¤‘...")

    sam.to(device=device)
    predictor = SamPredictor(sam)
    
    # 3. í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€ ë¡œë“œ
    image = cv2.imread("test_images/garment/test_garment.jpg")
    image_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    
    # 4. ì„¸ê·¸ë©˜í…Œì´ì…˜ ì‹¤í–‰
    print("ì„¸ê·¸ë©˜í…Œì´ì…˜ ì²˜ë¦¬ ì¤‘...")

    masks, scores, logits = predictor.predict(
        point_coords=input_point,
        point_labels=input_label,

    print(f"ì²˜ë¦¬ ì‹œê°„: {process_time:.2f}ì´ˆ")
    
    # 5. ê²°ê³¼ ì €ì¥ (ìµœê³  ì ìˆ˜ ë§ˆìŠ¤í¬)

    # ë§ˆìŠ¤í¬ë¥¼ ì›ë³¸ ì´ë¯¸ì§€ì— ì ìš©

    print("ê²°ê³¼ ì €ì¥ ì™„ë£Œ")
    print(f"ë§ˆìŠ¤í¬ í’ˆì§ˆ ì ìˆ˜: {scores[best_mask_idx]:.3f}")
    
    return {
        "process_time": process_time,
        "mask_score": float(scores[best_mask_idx]),
        "result_path": "results/sam_mask.png"
    }

if __name__ == "__main__":
    results = test_sam()
    if results:
        print("\n=== í…ŒìŠ¤íŠ¸ ê²°ê³¼ ===")
        for key, value in results.items():

# test_mediapipe.py
import cv2

    # 1. MediaPipe ì´ˆê¸°í™”
    mp_face_mesh = mp.solutions.face_mesh

        min_detection_confidence=0.5

    # 2. í…ŒìŠ¤íŠ¸ ì´ë¯¸ì§€ ë¡œë“œ
    image = cv2.imread("test_images/person/test_face.jpg")

    # 3. ì–¼êµ´ ëœë“œë§ˆí¬ ì¶”ì¶œ
    print("ì–¼êµ´ ëœë“œë§ˆí¬ ì¶”ì¶œ ì¤‘...")

    process_time = time.time() - start_time

    # 4. ëœë“œë§ˆí¬ ì‹œê°í™” ë° ì €ì¥

            annotated_image,

            None,
            mp_drawing_styles.get_default_face_mesh_contours_style()

        # ëœë“œë§ˆí¬ ì¢Œí‘œ ì¶”ì¶œ
        landmarks = []

                landmark.x * w,
                landmark.y * h,
                landmark.z
            ])
        
        print(f"ëœë“œë§ˆí¬ ê°œìˆ˜: {len(landmarks)}")
        print(f"ê²°ê³¼ ì €ì¥: results/mediapipe_landmarks.jpg")
        
        return {
            "process_time": process_time,
            "landmark_count": len(landmarks),
            "landmarks": np.array(landmarks),
            "result_path": "results/mediapipe_landmarks.jpg"
        }
    else:
        print("âš ï¸ ì–¼êµ´ì„ ê°ì§€í•˜ì§€ ëª»í–ˆìŠµë‹ˆë‹¤.")
